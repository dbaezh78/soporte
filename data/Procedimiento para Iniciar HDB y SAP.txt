*su - hdbadm*

# hdbadm@SAPQA:/usr/sap/HDB/HDB02> HDB start

*HDB start*

*exit*

# Luego que salgamos como comentamos anteriormente, entraremos con el usuario "acqadm" usando el privilegio su.
# SAPQA:~ # su - acqadm

*su - acqadm*

# ahora iniciamos el SAP con el siguiente comando startsap
# SAPQA:acqadm 51> startsap

*startsap*
